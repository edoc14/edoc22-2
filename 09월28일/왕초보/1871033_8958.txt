import java.util.Scanner;public class Main {	public static void main(String[] args) {		// TODO Auto-generated method stub				Scanner scanner=new Scanner(System.in);		int cases=scanner.nextInt();				String q[]=new String[cases];		int score;		int sum[]=new int[cases];				for(int r=0;r<cases;r++) {			q[r]=scanner.next();						score=0;			sum[r]=0;						for(int i=0;i<q[r].length();i++) {				if(q[r].charAt(i)=='O') {					score++;					sum[r]+=score;				}				else score=0;			}					}				for(int r=0;r<cases;r++)			System.out.println(sum[r]);						scanner.close();	}}
